import { PrismaClient } from '@/prisma/generated/prisma-db-auth/client';
import { PrismaPg } from '@prisma/adapter-pg';

const adapter = new PrismaPg({
  connectionString: process.env.AUTH_DATABASE_URL!,
});

const globalForPrisma = global as unknown as {
  prisma: PrismaClient;
  prismaAuthInstances?: number;
};

if (!globalForPrisma.prismaAuthInstances) {
  globalForPrisma.prismaAuthInstances = 0;
}

globalForPrisma.prismaAuthInstances += 1;
console.log(`Prisma Auth instances created: ${globalForPrisma.prismaAuthInstances}`);

const dbAuth =
  globalForPrisma.prisma || new PrismaClient({
    adapter,
    log:
      process.env.NODE_ENV === 'development'
        ? ['query', 'error', 'warn']
        : ['error'],
  });

if (process.env.NODE_ENV !== 'production') globalForPrisma.prisma = dbAuth;

export default dbAuth;
