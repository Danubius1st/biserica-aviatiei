// Alternative 1: Node.js runtime with more location data
import { NextResponse } from 'next/server';

export const runtime = 'nodejs'; // switched to nodejs runtime

export async function GET(request: Request) {
  // Get multiple location headers from Vercel
  const city = request.headers.get('x-vercel-ip-city') || 'Unknown';
  const country = request.headers.get('x-vercel-ip-country') || 'Unknown';
  const region = request.headers.get('x-vercel-ip-country-region') || 'Unknown';
  const latitude = request.headers.get('x-vercel-ip-latitude') || null;
  const longitude = request.headers.get('x-vercel-ip-longitude') || null;

  return NextResponse.json({
    city,
    country,
    region,
    coordinates: latitude && longitude ? { latitude, longitude } : null,
    timezone: request.headers.get('x-vercel-ip-timezone') || null,
  });
}
